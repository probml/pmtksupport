	model
		{
			for( i in 1 : N ) {
				r[i] ~ dbin(p[i],n[i])
				b[i] ~ dnorm(0.0,tau)
				logit(p[i]) <- alpha0 + alpha1 * x1[i] + b[i]
			}
			alpha0 ~ dnorm(0.0,1.0E-6)
			alpha1 ~ dnorm(0.0,1.0E-6)
			# Choice of priors for random effects variance
			# Prior 1: uniform on SD
			sigma~ dunif(0,100)
			tau<-1/(sigma*sigma)

		}
